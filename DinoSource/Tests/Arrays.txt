var arr = [1, 2, 3];

print("Size: " + str(arr.size()));        //  3
arr.add(4);
print("After add: " + str(arr));            //  [1, 2, 3, 4]
arr.remove(1);                               // Removes index 1 (number 2)
print("After remove: " + str(arr));          //  [1, 3, 4]
print("Has 3? " + str(arr.has(3)));         //  true
print("Index of 4: " + str(arr.find(4)));   //  2

var last = arr.pop_back();
print("Pop: " + str(last));                 //  4
print("After pop: " + str(arr));            //  [1, 3]

var copy = arr.copy();
copy.add(99);
print("Original: " + str(arr));            //  [1, 3]
print("Copy: " + str(copy));               //  [1, 3, 99]

arr.clear();
print("After clear: " + str(arr.size()));   //  0
